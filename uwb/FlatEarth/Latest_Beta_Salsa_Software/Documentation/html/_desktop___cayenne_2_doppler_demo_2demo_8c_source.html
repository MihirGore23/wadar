206 void interpretDoppler(char *text, double max, double distance, int *velocities, int len, int dopplerLen)
207 {
208  int i;
209 
210  // Compute average column index
211  int total = 0;
212  double ave;
213  for (i = 0; i < len; i++) {
214  total += velocities[i];
215  }
216  ave = 1.0 * total / len;
217 
218  if ((double)abs(max) > threshold) {
219  char *trend = getVelocityStr(ave, dopplerLen);
220  snprintf(text, 80, "[thres: %8.0lf] Target at %f m %s\n", threshold, distance, trend);
221  }
222  else {
223  snprintf(text, 80, "[thres: %8.0lf] No movement detected...", threshold);
