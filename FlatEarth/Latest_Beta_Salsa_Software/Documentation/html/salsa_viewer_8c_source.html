108 void update_ascan(SalsaViewer viewer, float *data)
109 {
110  int i;
111  for (i = 0; i < viewer->sigLength; i++) {
112  viewer->ascan[i].position.x = (float)i * viewer->ascan_xscale_mult;
113  viewer->ascan[i].position.y = data[i] * viewer->ascan_yscale_mult + viewer->ascan_yscale_offset;
114 
115  // Check boundary (set vertical position at ends if out-of-bounds)
116  if (viewer->ascan[i].position.y < 0) viewer->ascan[i].position.y = 0;
117  if (viewer->ascan[i].position.y > (viewer->winSize.y - TEXT_FONT_SAFE)) viewer->ascan[i].position.y = viewer->winSize.y - TEXT_FONT_SAFE;
118  }
119 }

131 void update_bscan(SalsaViewer viewer, float *data)
132 {
133  int i, j;
134 
135  // Update one row (random row color for now...)
136  for (i = 0, j = 0; i < 4 * viewer->sigLength; i += 4, j++) {
137  // Compute color based on data and bscan y-axis scaling factor
138  sfUint8 color = (sfUint8)(data[j] * viewer->bscan_yscale_mult + viewer->bscan_yscale_offset);
139 
140  // Colors can range from 0 to 255 (so check upper bound)
141  if (color > 255) color = 255;
142 
143  // RGB == same color means grey-scale
144  viewer->bscan[i + 0] = color;
145  viewer->bscan[i + 1] = color;
146  viewer->bscan[i + 2] = color;
147 
148  // Set the alpha channel to 100% opaque
149  viewer->bscan[i + 3] = 255;
150  }
151 }

163 void update_tracker(SalsaViewer viewer, int trackIdx, sfColor color)
164 {
165  int i;
166 
167  // Latest update goes into index 0 -- the top of the window
168  viewer->tracker[0].position.x = (float)trackIdx * viewer->spriteScale.x;
169  viewer->tracker[0].position.y = 0;
170  viewer->tracker[0].color = color;
171 
172  // Update the y coordinates since we've shifted down
173  for (i = 1; i < viewer->bscanLength; i++) {
174  // The x coordinate will be the same
175  viewer->tracker[i].position.y = (float)i * viewer->spriteScale.y;
176  }
177 
178  // Iterate through entire signal and if on a boundary with index 0, then make
179  // the current alpha channel == 0. This just makes things look nicer, since
180  // the line draw might attempt to interpolate the alpha channel from 0 to x,
181  // which creates these ugly horizontal fades
182  for (i = 1; i < viewer->bscanLength - 2; i++) {
183  if ((viewer->tracker[i - 1].position.x == 0) || (viewer->tracker[i + 1].position.x == 0)) {
184  viewer->tracker[i].color.a = 0;
185  }
186  }
187 }

190 void search_min_max(SalsaViewer viewer, float *max, float *min)
191 {
192  int i;
193 
194  // Init the max/min values to the initial data sample
195  *max = *min = viewer->data[0];
196

198  for (i = 1; i < viewer->sigLength; i++) {
199  if (viewer->data[i] > *max) *max = viewer->data[i];
200  if (viewer->data[i] < *min) *min = viewer->data[i];
201  }

345

348  sfRenderWindow_close(viewer->window);

350

353  viewer->paused = !viewer->paused;
354  }
355 
356  // Toggle A-Scan if user hits [1] key

358  viewer->showAscan = !viewer->showAscan;
359  }
360

363  viewer->showBscan = !viewer->showBscan;
364  }
365 
366  // Reset vertical scale if user hits [R] key

408 void SalsaViewer_resetVScale(SalsaViewer viewer, float fmax, float fmin)
409 {
410  if (viewer != NULL) {
411  // Just bail if the max isn't greater than the min
412  if (fmax <= fmin) return;
413 
414  // Adjust fmax/fmin a little to give some vertical headroom each way
415  fmax *= 1.05;
416  fmin *= fmin > 0 ? 0.95 : 1.05;
417 
418  // Compute the vertical scaling factor and offset for A-Scan
419  viewer->ascan_yscale_mult = -1.0 * (viewer->winSize.y - TEXT_FONT_SAFE) / (fmax - fmin);
420  viewer->ascan_yscale_offset = -1.0 * fmax * viewer->ascan_yscale_mult;
421 
422  // Compute the vertical scaling factor and offset for B-Scan
423  viewer->bscan_yscale_mult = -1.0 * 255 / (fmax - fmin);
424  viewer->bscan_yscale_offset = -1.0 * fmax * viewer->bscan_yscale_mult;
425  }
426 }

429 void SalsaViewer_updateData(SalsaViewer viewer, float *data)
430 {
431  // Color doesn't really matter, but alpha channel should be 0!
432  sfColor nullColor = sfColor_fromRGBA(255, 255, 255, 0);
433 
434  if (viewer != NULL) {
435  // Copy the data array
436  memcpy(viewer->data, data, sizeof (float) * viewer->sigLength);
437 
438  // Update the A-Scan view
439  update_ascan(viewer, data);
440 
441  // Shift out old B-Scan data (old data will be overwritten)
442  shift_bscan(viewer);
443 
444  // Shift out old tracking line
445  shift_tracker(viewer);
446 
447  // Update B-Scan with current signal with no tracking line update
448  update_bscan(viewer, data);
449 
450  // Update the tracking line
451  update_tracker(viewer, 0, nullColor);
452 
453  // Update the texture (which is what is actually drawn)
454  sfTexture_updateFromPixels(viewer->texture, viewer->bscan, viewer->sigLength, viewer->bscanLength, 0, 0);
455  }
