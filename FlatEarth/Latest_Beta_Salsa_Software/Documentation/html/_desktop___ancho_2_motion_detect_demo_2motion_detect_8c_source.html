208 void radarClient_update(SalsaViewer viewer, char *text, uint32_t *counters)
209 {
210  int i;
211 
212  // Compute the current movement threshold (user can adjust the movement factor)
213  float threshold = YMAX * movementFactor;
214 
215  // Correct endianess of received radar counters
216  for (i = 0; i < numberOfSamplers; i++) counters[i] = ntohl(counters[i]);
217 
218  // Apply MTI clutter map
219  clutter_mti_process(counters, radarClutterRemoved, &mtiClutter);
220 
221  for (i = 0; i < numberOfSamplers; i++) {
222  // The signal to plot is just the counters with the clutter removed
223  plotSig[i] = radarClutterRemoved[i];
224 
225  // Our signal to process is the abs() of the clutter removed. In the next
226  // step, we do a threshold check
227  radarSig[i] = abs(radarClutterRemoved[i]);
228  }
229 
230  // Search forward for the first value above the threshold
231  for (i = 0; i < numberOfSamplers; i++) {
232  if (radarSig[i] > threshold) {
233  //printf("target at i = %d, %f\n", i, radarSig[i]);
234  break;
235  }
236  }
237 
238  // If we've found a target it will be smaller than the max...
239  if (i < numberOfSamplers) {
240  // Computed distance to target
241  float distance = offsetDistance + sampleResolution * i;
242 
243  // Update the viewer (add tracking line for target index)
244  snprintf(text, 80, "[ Threshold %.2f%% ] Movement at %.2f m", movementFactor * 100.0, distance);
245  SalsaViewer_updateDataPlusTrack(viewer, plotSig, i, sfColor_fromRGB(0, 255, 0));
246  }
247  else {
248  // Update the viewer (no tracking line)
249  snprintf(text, 80, "[ Threshold %.2f%% ] No movement", movementFactor * 100.0);
250  SalsaViewer_updateData(viewer, plotSig);
251  }
252 
253  SalsaViewer_updateText(viewer, text);
254 }
