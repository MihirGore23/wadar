349 void radarClient_update(SalsaViewer viewer, char *text, uint32_t *counters)
350 {
351  int i;
352 
353  // Correct endianess of received radar counters
354  for (i = 0; i < numberOfSamplers; i++) counters[i] = ntohl(counters[i]);
355 
356  // On our first frame, set the static cluttermap
357  if (isFirstFrame) {
358  clutter_setSCluttermap(counters, &staticClutter);
359  isFirstFrame = false;
360  // Skip the rest of this iteration
361  return;
362  }
363 
364  // Update the clutter map and remove clutter
365  if (useAdaptiveClutterMap) {
366  clutter_updateACluttermap(counters, &adaptiveClutter);
367  clutter_removeAClutter(counters, radarClutterRemoved, &adaptiveClutter);
368  }
369  else if (useMTIClutterMap) {
370  clutter_mti_process(counters, radarClutterRemoved, &mtiClutter);
371  }
372  else {
373  clutter_removeSClutter(counters, radarClutterRemoved, &staticClutter);
374  }
375 
376  for (i = 0; i < numberOfSamplers; i++) {
377  plotSig[i] = radarClutterRemoved[i];
378  }
379 
380  SalsaViewer_updateData(viewer, plotSig);
381 
382  switch (clutterMethod) {
383  case CLUTTERMAP_STATIC:
384  snprintf(text, 80, "[ Static ]");
385  break;
386 
387  case CLUTTERMAP_ADAPTIVE:
388  snprintf(text, 80, "[ Adaptive ] [alpha: %.2f beta: %.2f]", alpha, beta);
389  break;
390 
391  case CLUTTERMAP_MTI_3:
392  snprintf(text, 80, "[ MTI 3-pulse ]");
393  break;
394 
395  case CLUTTERMAP_MTI_5:
396  snprintf(text, 80, "[ MTI 5-pulse ]");
397  break;
398  }
399 
400  SalsaViewer_updateText(viewer, text);
401 }
