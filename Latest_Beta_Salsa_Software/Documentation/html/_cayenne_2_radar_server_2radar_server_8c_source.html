189 void handleClient(int sock)
190 {
191  uint8_t cmdBuffer[RCV_CMD_SIZE];
192  uint8_t infoBuffer[INFO_REQ_SIZE];
193  uint8_t *dataBuffer;
194 
195  // Radar stats
196  int numberOfSamplers;
197  float resolution;
198  float offsetDistance;
199 
200  // Pointers to arrays for signal storage
201  uint32_t *radarCounters;
202 
203  // Flags
204  bool sendInfoRequest = false;
205  bool sendCalibrating = false;
206  bool dataSendEnabled = false;
207 
208  // Loop counter
209  int i;
210 
211  // Timeout for recv() calls
212  struct timeval tv;
213  tv.tv_sec = 0;
214  tv.tv_usec = 10000; // 10 ms
215  setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, (char *)&tv, sizeof (struct timeval));
216 
217  // Determine the radar stats
218  numberOfSamplers = getIntValueByName(rh, "SamplersPerFrame");
219  resolution = (0.5 * C) / getFloatValueByName(rh, "SamplesPerSecond");
220  offsetDistance = getFloatValueByName(rh, "OffsetDistanceFromReference");
221 
222  // Allocate memory for signal storage
223  radarCounters = (uint32_t *)malloc(numberOfSamplers * sizeof (uint32_t));
224 
225  // Encode the info buffer
226  memset(infoBuffer, 0, sizeof (infoBuffer));
227  encodeInfoRequest(infoBuffer, numberOfSamplers, resolution, offsetDistance);
228 
229  // Have the data buffer also point to radar signal
230  dataBuffer = (uint8_t *)radarCounters;
231 
232  while (1) {
233  // Note: In this implementation, user commands are all 8 bytes in length,
234  // but the data *is* encoded into the <header><data> format. The <header>
235  // contains the command code, the data length (=4). The <data> is 4 bytes
236  // but is not used for anything.
237 
238  // See if there is more data to receive
239  int bytesRead = recv(sock, cmdBuffer, RCV_CMD_SIZE, 0);
240 
241  if (bytesRead == 0) {
242  // No bytes read means client closed the socket
243  break;
244  }
245  else if (bytesRead < 0) {
246  // EWOULDBLOCK or EAGAIN mean read timed out that's ok
247  if ((errno == EWOULDBLOCK) || (errno == EAGAIN)) {
248  // Do nothing
249  }
250  else {
251  // Exit loop on other error codes
252  break;
253  }
254  }
255  else {
256  // The first two bytes are the <header> command code (handle endianness)
257  uint16_t cmd = ntohs(unpackUInt16(cmdBuffer));
258 
259  // Use lookup table to determine what to do...
260  switch (cmd) {
261  case SERVER_INFO_REQ:
262  sendInfoRequest = true;
263  break;
264 
265  case CALIBRATE_START:
266  sendCalibrating = true;
267  break;
268 
269  case RADAR_START:
270  dataSendEnabled = true;
271  break;
272 
273  case RADAR_STOP:
274  dataSendEnabled = false;
275  break;
276  }
277  }
278 
279  if (sendInfoRequest) {
280  // Reset flag so we don't resend INFO each pass
281  sendInfoRequest = false;
282 
283  SalsaComm_send(sock, SERVER_INFO_RSP, INFO_REQ_SIZE, infoBuffer);
284  }
285 
286  if (sendCalibrating) {
287  // Reset flag so we don't calibrate each pass
288  sendCalibrating = false;
289 
290  // Send message indicating calibration started
291  SalsaComm_send(sock, CALIBRATE_START, 0, NULL);
292 
293  status = radarHelper_doAction(rh, "MeasureAll");
294 
295  // Send message indicating calibration done
296  SalsaComm_send(sock, CALIBRATE_DONE, 0, NULL);
297  }
298 
299  if (dataSendEnabled) {
300  status = radarHelper_getFrameRaw(rh, radarCounters, numberOfSamplers);
301 
302  // Correct endianness of data before sending
303  for (i = 0; i < numberOfSamplers; i++) radarCounters[i] = htonl(radarCounters[i]);
304 
305  // Send the latest radar signal
306  SalsaComm_send(sock, RADAR_DATA, numberOfSamplers * sizeof (uint32_t), dataBuffer);
307  }
308 
309  // Break out of loop if radar returned non-zero status
310  if (status) {
311  printf("Radar error occurred! err = [%d]\n", status);
312  break;
313  }
314 
315  usleep(1000);
316  }
317 
318  // Close client socket
319  close(sock);
320 
321  // Free memory allocated for radar signals
322  free (radarCounters);
323 }
