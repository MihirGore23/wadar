251 void doppler_interpret(double distance, int *velocities, int len, int doppler_len)
252 {
253  int i;
254 
255  static double last_distance = -1; // init to some impossible distance
256 
257  // Compute average column index
258  int total = 0;
259  double ave;
260  for (i = 0; i < len; i++) {
261  total += velocities[i];
262  }
263  ave = 1.0 * total / len;
264 
265  // Display ZONE info
266 
267  if (distance < zone1_2) {
268  printf("%s distance = %6.3f m [ 1 ]", ANSI_COLOR_RED, distance);
269  }
270  else if (distance < zone2_3) {
271  printf("%s distance = %6.3f m [ 2 ]", ANSI_COLOR_YELLOW, distance);
272  }
273  else {
274  printf("%s distance = %6.3f m [ 3 ]", ANSI_COLOR_GREEN, distance);
275  }
276 
277  // Note: The comparison thresholds used below are arbitrary. They were chosen
278  // to achieve overall smooth trending, but they miss some slower movement...
279 
280  // Display message based on average column index
281  if (ave < 0.4 * doppler_len && distance > last_distance) {
282  printf("%s target moving away...", ANSI_COLOR_MAGENTA);
283  }
284  else if (ave > 0.6 * doppler_len && distance < last_distance) {
285  printf("%s target moving toward...", ANSI_COLOR_CYAN);
286  }
287 
288  printf(ANSI_COLOR_RESET "\n");
289 
290  // Store distance for next time comparison (not filtered...)
291  last_distance = distance;
292 
293  // An improvement here might be to not only filter the distance, but also the
294  // last distance? The check in the if() above is just to not choose an "alias"
295  // trend. In the Doppler, we can get some wrapping over, so this acts like a
296  // sanity-check.
297 
298  // Even with our distance check, the fact that Doppler aliasing is occurring
299  // can cause the target moving away/toward to be in wrong direction!
300 }
