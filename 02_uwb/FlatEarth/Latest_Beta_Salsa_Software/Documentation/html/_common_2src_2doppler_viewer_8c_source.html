151 void rotate(double *dopplerSrc, double *dopplerDst, int rowLen, int colLen)
152 {
153  int row;
154  int dstCol, srcCol;
155  for (srcCol = colLen / 2, dstCol = 0; srcCol < colLen; srcCol++, dstCol++) {
156  for (row = 0; row < rowLen; row++) {
157  dopplerDst[row * colLen + dstCol] = dopplerSrc[row * colLen + srcCol];
158  }
159  }
160  for (srcCol = 0; srcCol < colLen / 2; srcCol++, dstCol++) {
161  for (row = 0; row < rowLen; row++) {
162  dopplerDst[row * colLen + dstCol] = dopplerSrc[row * colLen + srcCol];
163  }

245 void update_doppler_view(DopplerViewer viewer, double *data)
246 {
247  int i, r, c;
248 
249  // Update one row (random row color for now...)
250  for (i = 0, r = 0, c = 0; i < 4 * viewer->sigLength * viewer->fftLength; i += 4) {
251  // Compute color based on data and bscan y-axis scaling factor
252  sfUint8 color = (sfUint8)(abs(data[r * viewer->fftLength + c]) * viewer->yscale_mult + viewer->yscale_offset);
253 
254  // Colors can range from 0 to 255 (so check upper bound)
255  if (color > 255) color = 255;
256 
257  // RGB == same color means grey-scale
258  viewer->dopplerData[i + 0] = color;
259  viewer->dopplerData[i + 1] = color;
260  viewer->dopplerData[i + 2] = color;
261 
262  // Set the alpha channel to 100% opaque
263  viewer->dopplerData[i + 3] = 255;
264 
265  c++;
266  if (c == viewer->fftLength) {
267  c = 0;
268  r++;
269  }
270  }
271 }

477 void DopplerViewer_resetVScale(DopplerViewer viewer, float fmax, float fmin)
478 {
479  if (viewer != NULL) {
480  // Just bail if the max isn't greater than the min
481  if (fmax <= fmin) return;
482 
483  // Adjust fmax/fmin a little to give some vertical headroom each way
484  fmax *= 1.05;
485  fmin *= fmin > 0 ? 0.95 : 1.05;
486 
487  // Compute the vertical scaling factor and offset for B-Scan
488  viewer->yscale_mult = -1.0 * 255 / (fmax - fmin);
489  viewer->yscale_offset = -1.0 * fmax * viewer->yscale_mult;
490  }

494 void DopplerViewer_updateData(DopplerViewer viewer, uint32_t *counters)
495 {
496  if (viewer != NULL) {
497  int i;
498  // Correct endianess of received radar counters
499  for (i = 0; i < viewer->sigLength; i++) counters[i] = ntohl(counters[i]);
500 
501  // Apply the MTI filter
502  clutter_mti_process(counters, viewer->radarClutterRemoved, &viewer->mtiClutter);
503 
504  // Convert signal with clutter removed into floating-point signal
505  radarDSP_convertIntArray(viewer->radarClutterRemoved, viewer->radarSignalReal, viewer->sigLength);
506 
507  // Compute analytic signal (I & Q)
508  radarDSP_dp_Hilbert(viewer->radarSignalReal, viewer->radarSignalImag, viewer->sigLength);
509 
510  // Add envelope to the doppler matrix
511  doppler_addFrame(viewer->radarSignalReal, viewer->radarSignalImag, &viewer->doppler);
512  }
513 }

524 void DopplerViewer_process(DopplerViewer viewer)
525 {
526  if (viewer != NULL) {
527  double max;
528  int maxIdx;
529 
530  // Process doppler matrix
531  doppler_processWindow(&viewer->doppler);
532  doppler_getLastWindow(viewer->dopplerMatrixReal, viewer->dopplerMatrixImag, &viewer->doppler);
533 
534  // Reorder matrix to match what would happen in MATLAB (DC in center)
535  rotate(viewer->dopplerMatrixReal, viewer->dopplerMatrixReordered, viewer->sigLength, viewer->fftLength);
536 
537  // Interpret the Doppler matrix
538  radarDSP_dp_Max(viewer->dopplerMatrixReordered, viewer->sigLength * viewer->fftLength, &max, &maxIdx);
539  // Determine row, col which stores distance bin, velocity
540  viewer->maxVal = max;
541  viewer->maxRow = maxIdx / viewer->fftLength;
542  viewer->maxCol = maxIdx % viewer->fftLength;
543 
544  // Update the doppler matrix
545  update_doppler_view(viewer, viewer->dopplerMatrixReordered);
546 
547  // Update the texture (which is what is actually drawn)
548  sfTexture_updateFromPixels(viewer->texture, viewer->dopplerData, viewer->fftLength, viewer->sigLength, 0, 0);
549  }
550 }
